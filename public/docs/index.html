    Retrieve all branch's bookings. Example: /api/branches/1/bookings?start=2021-01-24&end=2021-01-30

    Example request:

    curl -X GET \
    -G "https://it114118-fyp.herokuapp.com/api/branches/{branch}/bookings?start=tempore&end=dolor" \
    -H "Content-Type: application/json" \
    -H "Accept: application/json"
    const url = new URL(
    "https://it114118-fyp.herokuapp.com/api/branches/{branch}/bookings"
);

let params = {
    "start": "tempore",
    "end": "dolor",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

    "Content-Type": "application/json",
    "Accept": "application/json",

    method: "GET",
